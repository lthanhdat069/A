-- Tạo GUI chính
local screenGui = Instance.new("ScreenGui")
screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

-- Tạo khung menu (Frame)
local frame = Instance.new("Frame")
frame.Parent = screenGui
frame.Size = UDim2.new(0, 250, 0, 150)
frame.Position = UDim2.new(0.5, -125, 0.5, -75)
frame.BackgroundColor3 = Color3.fromRGB(0, 200, 255)
frame.Visible = false -- ẩn menu lúc đầu
local toggle = Instance.new("TextButton")
toggle.Parent = screenGui
toggle.Text = "Mở Menu"
toggle.Size = UDim2.new(0, 120, 0, 40)
toggle.Position = UDim2.new(0.05, 0, 0.8, 0)
toggle.BackgroundColor3 = Color3.fromRGB(0, 255, 150)
toggle.MouseButton1Click:Connect(function()
	-- nếu menu đang ẩn thì bật, còn đang bật thì ẩn
	if frame.Visible == false then
		frame.Visible = true
		toggle.Text = "Chết"
	else
		frame.Visible = false
		toggle.Text = "Sống"
	end
end)
